#include <iostream>
#include <conio2.h>
#include <ctime>

using namespace std;

int main(int argc, char *argv[]) {
	
	// Declaracion e inicializacion de variables 	
	int monedas=0;
	bool play=true;
	bool lose=false;
	bool bandera=true;
	char tecla=' ';
	int contAnimacion=0;
	
	srand(time(NULL));	
	int n1=0;
	int n2=0;
	int n3=0;
	
	string header1="* * * * * * * * * * * * * * * * * * * * * * * *";
	string header2="*       P       |        I        |     E     *";
	string header3="* Tirar palanca | Insertar moneda | Retirarse *";
	string header4="*  El juego permite ingresar hasta 9 monedas  *";
	string header5="*                                             *";
	
	string linea1="*                   \\|||/                     *";
	string linea2="*                   (   )                     *";
	string linea3="*                    \\ /                      *";
	string linea4="*          .----oo0--(_)---------.            *";
	string linea5="*          | Coins ";
	string linea5b="             |            *";
	string linea6="*          |                     |            *";
	string linea7="*          |    .-----------.    |            *";
	string linea8="*          |    | ";
	string barraVertical=" | ";
	string linea8b="   |            *";	
	string linea9="*          |    .-----------.    |            *";	
	string linea10="*          |                     |            *";					
	string linea11="*          |                     |            *";
	string linea12="*          :---------------oo0---:            *";
	string linea13="*                  |  |  |                    *";
	string linea14="*                  |  |  |                    *"; 
	string linea15="*                  |__|__|                    *";
	string linea16="*                ___|| ||___                  *";
	string linea17="*               (____] [____)                 *";
	string linea18="* * * * * * * * * * * * * * * * * * * * * * * *";
	
	// ****** Calibracion del reloj *****//
	gotoxy(10,10);
	cout<<"CARGANDO...ESPERE UNOS INSTANTES...";
	
	int t1, t2;	
	int cont=0;
	int ciclos=0;
	int sumaCiclos=0;
	int promedioCiclos=0;
	int contCiclos=0;
	
	double delay=0.4;
	int referenciaCiclos=0;
	
	while (cont<10){
		
		t1=time(NULL);
		t2=time(NULL);
		
		while (t1==t2){
			
			t2=time(NULL);
			ciclos++;
			
		}
		
		if (cont>0){
			
			sumaCiclos+=ciclos;
		}
		
		cont++;
		ciclos=0;
	}
	
	promedioCiclos=sumaCiclos/(cont-1);	
	referenciaCiclos=promedioCiclos*delay;	
	
	// Se muestran los headers principales
	gotoxy(1,1);

	cout<<header1<<endl;
	cout<<header2<<endl;
	cout<<header3<<endl;
	cout<<header4<<endl;
	cout<<header5<<endl;
	
	//Loop principal del juego
	while (play){
		
		gotoxy(1,6);
		
		cout<<linea1<<endl;
		cout<<linea2<<endl;	
		cout<<linea3<<endl;
		cout<<linea4<<endl;		
		cout<<linea5<<monedas<<linea5b<<endl;
		cout<<linea6<<endl;
		cout<<linea7<<endl;
		cout<<linea8<<n1<<barraVertical<<n2<<barraVertical<<n3<<barraVertical<<linea8b<<endl;
		cout<<linea9<<endl;
		cout<<linea10<<endl;
		cout<<linea11<<endl;
		cout<<linea12<<endl;
		cout<<linea13<<endl;
		cout<<linea14<<endl;
		cout<<linea15<<endl;
		cout<<linea16<<endl;
		cout<<linea17<<endl;
		cout<<linea18<<endl;
		
		if(lose){
			
			play=false;
		}
		
		if (kbhit() || !bandera){
			
			if (bandera){
				
				tecla=getch();	
			} 
			
			switch(tecla){
				
			//Animacion del egreso de monedas
			case 'E':
				
				if (monedas>0){
				
					if (contAnimacion==0){
						
						bandera=false;
						linea13=linea13.replace(30,1,"*");
						
					} if (contAnimacion==1){
						
						linea13=linea13.replace(30,1," ");
						linea14=linea14.replace(30,1,"*");					
						
					} if (contAnimacion==2){
						
						linea14=linea14.replace(30,1," ");
						linea15=linea15.replace(30,1,"*");
						
					} if (contAnimacion==3){
						
						linea15=linea15.replace(30,1," ");
						linea16=linea16.replace(30,1,"*");
						
					} if (contAnimacion==4){
						
						linea16=linea16.replace(30,1," ");
						linea17=linea17.replace(30,1,"*");
						
					} if (contAnimacion==5){
						
						linea17=linea17.replace(30,1," ");
						monedas--;

						contAnimacion=-1;	
					}	
					
					contAnimacion++;
					break;
				} 
				
				lose=true;
				break;
				
			//Animacion del ingreso de monedas	
			case 'I':
		
				if (contAnimacion==0){
					
					bandera=false;
					linea1=linea1.replace(29,1,"*");
					
				} if (contAnimacion==1){
					
					linea1=linea1.replace(29,1," ");
					linea2=linea2.replace(29,1,"*");					
					
				} if (contAnimacion==2){
					
					linea2=linea2.replace(29,1," ");
					linea3=linea3.replace(29,1,"*");
					
				} if (contAnimacion==3){
					
					linea3=linea3.replace(29,1," ");
					linea4=linea4.replace(29,1,"*");
					
				} if (contAnimacion==4){
					
					linea4=linea4.replace(29,1,"-");
					
					if(monedas<9){
						
						monedas++;								
						
					} else {
						
						linea10=linea10.replace(16,13,"I AM FULL!!!!");
					}
					
					bandera=true;
					contAnimacion=-1;					
				}
				
				contAnimacion++;
				break;
			
			//Animacion de la palanca
			case 'P':
				
				if (monedas>0){
					
					n1=rand()%7+1;
					n2=rand()%7+1;
					n3=rand()%7+1;
					
					if (contAnimacion==0){

						bandera=false;
						
						linea2=linea2.replace(21,1,"o");
						linea2=linea2.replace(23,1,"o");
						
						linea10=linea10.replace(16,13,"             ");
						
					}
					
					if (contAnimacion==29){
						
						monedas--;
						
						if(n1==n2 && n2==n3){
							
							int sumaMonedas=(n1+n2+n3);
							monedas+=sumaMonedas;
							
							linea2=linea2.replace(21,1,"$");
							linea2=linea2.replace(23,1,"$");
							
							linea10=linea10.replace(16,13,"YOU WIN!!!!!!");
							
							
						} else {
							
							linea2=linea2.replace(21,1,"X");
							linea2=linea2.replace(23,1,"X");
							
							if (monedas==0){
								
								lose=true;
								linea10=linea10.replace(16,13,"YOU LOSE!!!!!");
							}
						}
						
						bandera=true;
						contAnimacion=-1;					
					}
				}
				
				contAnimacion++;
				break;

			}
		}
		
		//DELAY
		while (contCiclos<referenciaCiclos){			
			
			contCiclos++;
		}
		
		ciclos=0;
		contCiclos=0;
		
	}
	
	return 0;
}
